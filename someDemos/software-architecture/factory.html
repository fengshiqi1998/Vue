class语法糖实现 
        class Factory {
          createOS() {
            console.log('undefined OS');
            throw new Error('can not dispatch function in class Factory directory.');
          }
        }
        class Android extends Factory {
          // 不重写就会使用Factory的方法
          createOS() {
            console.log('Android OS');
          }
        }
ES5实现 
        function Factory() {
          this.createOS = function() {
            console.log('undefined OS');
            throw new Error('can not dispatch function in class Factory directory.');
          }
        }
        function Android() {
          Factory.call(this);
          this.createOS = function() {
            console.log('Android OS');
          }
        }
